server {
    server_name booth-tool.secret-sealing.club www.booth-tool.secret-sealing.club;

    location / {
        root /root/abl-booth-tool/frontend/dist;
        try_files $uri $uri/ /index.html;
    }

    location /sale {
        proxy_pass http://unix:/run/abl-booth-tool.sock; 
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location /static {
        alias /root/abl-booth-tool/backend/static;
        expires 7d;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/booth-tool.secret-sealing.club-0001/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/booth-tool.secret-sealing.club-0001/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}
server {
    if ($host = www.booth-tool.secret-sealing.club) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = booth-tool.secret-sealing.club) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name booth-tool.secret-sealing.club www.booth-tool.secret-sealing.club;
    return 404; # managed by Certbot




}
