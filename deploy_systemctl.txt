[Unit]
Description=Gunicorn instance to serve abl-booth-tool
After=network.target

[Service]
User=root
Group=www-data
WorkingDirectory=/root/abl-booth-tool/backend
Environment="PATH=/root/abl-booth-tool/backend/venv/bin"
EnvironmentFile=/root/abl-booth-tool/backend/.env
ExecStart=/root/abl-booth-tool/backend/venv/bin/gunicorn --workers 3 --bind unix:/run/abl-booth-tool.sock run:app

[Install]
WantedBy=multi-user.target
